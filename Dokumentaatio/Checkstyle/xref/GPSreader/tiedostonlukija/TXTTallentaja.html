1   package GPSreader.tiedostonlukija;
2   
3   import java.io.FileWriter;
4   import java.io.IOException;
5   import GPSreader.sovelluslogiikka.Matka;
6   
7   /**
8    * Tallentaa jo ohjelmaan tuodun Matka-olion tekstimuotoiseksi tiedostoksi.
9    *
10   */
11  public class TXTTallentaja {
12  
13      /**
14       * Kirjoittaa Matka-olion tekstitiedostoon haluttuun tallennuskansioon.
15       * Tallennettu tiedosto luetaan TXTTallennettuLukija-luokalla.
16       *
17       * @param talletettavaMatka Matka-olio joka tallennetaan.
18       * @param tallennuskansio Kansio johon tiedosto tallennetaan
19       * @return Onnistuiko matkan tallentaminen
20       */
21      public boolean kirjoitaMatkaTiedostoon(Matka talletettavaMatka, String tallennuskansio) {
22          String nimi = talletettavaMatka.getMatkanNimi();
23          try {
24              FileWriter kirjoittaja = new FileWriter(tallennuskansio + "/" + nimi + ".txt");
25  
26              for (int i = 0; i < talletettavaMatka.getLatitudi().size(); i++) {
27                  String lat = Double.toString(talletettavaMatka.getLatitudi().get(i));
28                  String lon = Double.toString(talletettavaMatka.getLongitudi().get(i));
29                  String aika = String.valueOf(talletettavaMatka.getAikaleima().get(i).getTime());
30                  String tark = Double.toString(talletettavaMatka.getMittauksentarkkuus().get(i));
31  
32                  kirjoittaja.write(lat + "," + lon + "," + aika + "," + tark);
33                  kirjoittaja.write(System.lineSeparator());
34  
35              }
36              kirjoittaja.close();
37              return true;
38  
39          } catch (IOException ex) {
40              return false;
41  
42          }
43  
44      }
45  
46  }
